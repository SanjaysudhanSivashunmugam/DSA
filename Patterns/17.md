public class Main
{
	public static void main(String[] args) {
	    int n = 5;
	    int space = 2 * n - 2; 
	    for(int i = 0; i < 2 * n - 1; i++) {
	        int star = i + 1;
	        if(i >= n) {
	            star = 2 * n - i - 1;
	        }
	        
	        for(int j = 0; j < star; j++) {
	            System.out.print("*");
	        }
	        
	        for(int j = 0; j < space; j++) {
	            System.out.print(" ");
	        }
	        
	        if(i >= n - 1) {
	            space += 2;
	        }
	        
	        else {
	            space -= 2;
	        }
	        
	        for(int j = 0; j < star; j++) {
	            System.out.print("*");
	        }
	        
	        System.out.println();
	    }
	}
}
