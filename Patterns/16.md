```java
public class Main
{
	public static void main(String[] args) {
		int n = 5;
		
		//first half
		
		int space = 0;
		for(int i = 0; i < n; i++) {
		    for(int j = 0; j < n - i; j++) {
		        System.out.print("*");
		    }
		    
		    for(int j = 0; j < space; j++) {
		        System.out.print(" ");
		    }
		    space += 2;
		    
		    for(int j = 0; j < n - i; j++) {
		        System.out.print("*");
		    }
		    System.out.println();
		}
		// 2nd half
		space = 2 * (n - 1);
		for(int i = 0; i < n; i++) {
		    for(int j = 0; j <= i; j++) {
		        System.out.print("*");
		    }
		    
		    for(int j = 0; j < space; j++) {
		        System.out.print(" ");
		    }
		    space -= 2;
		    
		    for(int j = 0; j <= i; j++) {
		        System.out.print("*");
		    }
		    System.out.println();
		}
	}
}
```
