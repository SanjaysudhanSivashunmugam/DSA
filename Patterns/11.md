public class Main
{
	public static void main(String[] args) {
		int n = 4;
		int space = 2 * (n - 1);
		for(int i = 0; i < n; i++) {
		    for(int j = 0; j <= i; j++) {
		        System.out.print(j + 1);
		    }
		    
		    for(int j = 0; j < space; j++) {
		        System.out.print(" ");
		    }
		    space -= 2;
		    
		    for(int j = i + 1; j > 0; j--) {
		        System.out.print(j);
		    }
		    System.out.println();
		}
	}
}
